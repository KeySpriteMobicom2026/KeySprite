def visualize_waveform(song_path):
    import librosa
    import matplotlib.pyplot as plt
    y, sr = librosa.load(song_path)
    plt.figure(figsize=(14, 5))
    librosa.display.waveshow(y, sr=sr)
    plt.title('Waveform Visualization 🎧')
    plt.show()
