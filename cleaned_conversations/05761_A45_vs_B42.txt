```python
class CognitiveStyleGenerator(nn.Module):
    def __init__(self, style_dim=512, signal_length=256):
        super().__init__()
        self.style_encoder = nn.Sequential(
            spectral_norm(nn.Conv1d(1, 64, 5, stride=2)),
            nn.LeakyReLU(0.2),
            SelfAttention(64)
        )
        
        # 这里加入了一个可解释性层，用于可视化风格映射
        self.explainer = CognitiveMapper(style_dim)

    def forward(self, input_signal, return_style=False):
        style_code = self.style_encoder(input_signal)
        if return_style:
            return style_code, self.explainer(style_code)
        return self.generator(input_signal, style_code)
```
